\section*{Obligatoriske oppgaver}



\begin{oppgave}{E1}
La 
\[
A=
\begin{bmatrix}
1 & -1 & 1 \\ -2 & 5 & -4 \\ -1 & 1 & -1
\end{bmatrix}
\quad
\text{og }
\quad
b=
\begin{bmatrix}
28 \\ -20 \\ 10 
\end{bmatrix} 
\]
Finn alle løsninger av systemet $Ax=b$, 
og regn ut determinanten til $A$.\\
Er kolonnene i $A$ lineært uavhengige?
\end{oppgave}



\begin{oppgave}{E2}
Løs likningen $A\V{x}=\V{b}$ for 
$$
A=
	\begin{bmatrix}
	8  & -7 & 0 \\
	-8 & -7 & 3 \\
	-4 & 5  & -8\\
	-6 & 6  & -4
	\end{bmatrix}
\qquad 
\text{og}\quad
        \V{b}=
	\begin{bmatrix}
	-3  \\
	-7 \\
	-3 \\
	0
	\end{bmatrix}.
	$$
	Er kolonnene i $A$ lineært uavhengige?
\end{oppgave}
